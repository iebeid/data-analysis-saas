proc sql;
create table gtot as
select count(unique codename) as noo from survey
;
create table itot as
select answer, count(unique codename) as n1o from survey group by 1
;
create table ab as
select a.answer as answera, b.answer as answerb, count(*) as n11 from
	(select unique q_nbr,answer, codename from survey) a,
	(select unique q_nbr, answer, codename from survey) b
	where a.codename=b.codename
	and a.answer ne b.answer
	and a.q_nbr>b.q_nbr
	group by 1,2
;
select answera, answerb, (n11*n22)/(n21*n12) as oddsratio, n11, n1o, no1,noo
from
(select answera, answerb, n11, n1o, no1, noo, n1o-n11 as n12, no1-n11 as n21, noo-n11-(n1o-n11)-(no1-n11) as n22
from ab a,
itot b,
(select answer, n1o as no1 from itot) c,
gtot d
where b.answer=answera
and c.answer=answerb
) aa
having oddsratio ne .
and n11>=5
order by 3 desc
;
select answera, answerb, (n11*n22)/(n21*n12) as or,  
exp(log((n11*n22)/(n21*n12))+3*(((1/n11)+(1/n1o)+(1/no1)+(1/noo))**.5)) as uclor,
exp(log((n11*n22)/(n21*n12))-3*(((1/n11)+(1/n1o)+(1/no1)+(1/noo))**.5)) as lclor
from
(select answera, answerb, n11, n1o, no1, noo, n1o-n11 as n12, no1-n11 as n21, noo-n11-(n1o-n11)-(no1-n11) as n22
from ab a,
itot b,
(select answer, n1o as no1 from itot) c,
gtot d
where b.answer=answera
and c.answer=answerb
) aa
having or ne .
and n11>=5
order by 3 desc
;
quit;
